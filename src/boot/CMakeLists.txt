file(GLOB BOOT_ASM_DEPENDENCIES CONFIGURE_DEPENDS "${CMAKE_CURRENT_SOURCE_DIR}/*.asm")
list(SORT BOOT_ASM_DEPENDENCIES)

add_custom_command(
  OUTPUT "${OS1_BOOT_BIN}"
  COMMAND "${CMAKE_COMMAND}" -E make_directory "${OS1_ARTIFACT_DIR}"
  COMMAND "${OS1_NASM_EXECUTABLE}" -fbin "${CMAKE_CURRENT_SOURCE_DIR}/boot.asm" -o "${OS1_BOOT_BIN}"
  DEPENDS ${BOOT_ASM_DEPENDENCIES}
  WORKING_DIRECTORY "${CMAKE_CURRENT_SOURCE_DIR}"
  VERBATIM
)

add_custom_command(
  OUTPUT "${OS1_KERNEL16_BIN}"
  COMMAND "${CMAKE_COMMAND}" -E make_directory "${OS1_ARTIFACT_DIR}"
  COMMAND "${OS1_NASM_EXECUTABLE}" -fbin "${CMAKE_CURRENT_SOURCE_DIR}/kernel16.asm" -o "${OS1_KERNEL16_BIN}"
  DEPENDS ${BOOT_ASM_DEPENDENCIES}
  WORKING_DIRECTORY "${CMAKE_CURRENT_SOURCE_DIR}"
  VERBATIM
)

add_custom_target(boot_bin DEPENDS "${OS1_BOOT_BIN}")
add_custom_target(kernel16_bin DEPENDS "${OS1_KERNEL16_BIN}")
