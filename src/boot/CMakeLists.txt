set(BOOT_BIOS_DIR "${CMAKE_CURRENT_SOURCE_DIR}/bios")

file(GLOB BOOT_ASM_DEPENDENCIES CONFIGURE_DEPENDS "${BOOT_BIOS_DIR}/*.asm")
list(SORT BOOT_ASM_DEPENDENCIES)
set(BOOT_SHARED_DEPENDENCIES
  ${BOOT_ASM_DEPENDENCIES}
  "${PROJECT_SOURCE_DIR}/src/kernel/handoff/memory_layout.inc"
  "${OS1_KERNEL_LAYOUT_INC}"
  "${OS1_IMAGE_LAYOUT_INC}"
)

add_custom_command(
  OUTPUT "${OS1_BOOT_BIN}"
  COMMAND "${CMAKE_COMMAND}" -E make_directory "${OS1_ARTIFACT_DIR}"
  COMMAND "${OS1_NASM_EXECUTABLE}" "-I${OS1_GENERATED_DIR}/" "-I${BOOT_BIOS_DIR}/" -fbin "${BOOT_BIOS_DIR}/boot.asm" -o "${OS1_BOOT_BIN}"
  DEPENDS ${BOOT_SHARED_DEPENDENCIES}
  WORKING_DIRECTORY "${CMAKE_CURRENT_SOURCE_DIR}"
  VERBATIM
)

add_custom_command(
  OUTPUT "${OS1_KERNEL16_BIN}"
  COMMAND "${CMAKE_COMMAND}" -E make_directory "${OS1_ARTIFACT_DIR}"
  COMMAND "${OS1_NASM_EXECUTABLE}" "-I${OS1_GENERATED_DIR}/" "-I${BOOT_BIOS_DIR}/" -fbin "${BOOT_BIOS_DIR}/kernel16.asm" -o "${OS1_KERNEL16_BIN}"
  DEPENDS ${BOOT_SHARED_DEPENDENCIES}
  WORKING_DIRECTORY "${CMAKE_CURRENT_SOURCE_DIR}"
  VERBATIM
)

add_custom_target(boot_bin DEPENDS "${OS1_BOOT_BIN}")
add_custom_target(kernel16_bin DEPENDS "${OS1_KERNEL16_BIN}")
